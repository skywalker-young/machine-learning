class Node(object):
    def __init__(self,value=None):
        self.value=value
        self.left=None
        self.right=None


class Tree(object):
    def __init__(self):
        self.root=Node('root')
        self.size=1
        self.decision=['left','right']
        self.fitness=None

    def inorder(self,root):  # 中序序遍历
        if root is None:
            return []

        result = [root.value]
        left_item =  self.inorder(root.left)
        right_item =  self.inorder(root.right)
        return left_item + result + right_item

tree=Tree()
tree.root=Node('-')

tree.root.right=Node('*')
